32 static void simple_test_nfsft(void)
33 {
34  const int N = 4; /* bandwidth/maximum degree */
35  const int M = 8; /* number of nodes */
36  nfsft_plan plan; /* transform plan */
37  int j, k, n; /* loop variables */
38 
39  /* precomputation (for fast polynomial transform) */
40  nfsft_precompute(N,1000.0,0U,0U);
41 
42  /* Initialize transform plan using the guru interface. All input and output
43  * arrays are allocated by nfsft_init_guru(). Computations are performed with
44  * respect to L^2-normalized spherical harmonics Y_k^n. The array of spherical
45  * Fourier coefficients is preserved during transformations. The NFFT uses a
46  * cut-off parameter m = 6. See the NFFT 3 manual for details.
47  */
48  nfsft_init_guru(&plan, N, M, NFSFT_MALLOC_X | NFSFT_MALLOC_F |
49  NFSFT_MALLOC_F_HAT | NFSFT_NORMALIZED | NFSFT_PRESERVE_F_HAT,
50  PRE_PHI_HUT | PRE_PSI | FFTW_INIT | FFT_OUT_OF_PLACE, 6);
51 
52  /* pseudo-random nodes */
53  for (j = 0; j < plan.M_total; j++)
54  {
55  plan.x[2*j]= nfft_drand48() - K(0.5);
56  plan.x[2*j+1]= K(0.5) * nfft_drand48();
57  }
58 
59  /* precomputation (for NFFT, node-dependent) */
60  nfsft_precompute_x(&plan);
61 
62  /* pseudo-random Fourier coefficients */
63  for (k = 0; k <= plan.N; k++)
64  for (n = -k; n <= k; n++)
65  plan.f_hat[NFSFT_INDEX(k,n,&plan)] =
66  nfft_drand48() - K(0.5) + _Complex_I*(nfft_drand48() - K(0.5));
67 
68  /* Direct transformation, display result. */
69  nfsft_trafo_direct(&plan);
70  printf("Vector f (NDSFT):\n");
71  for (j = 0; j < plan.M_total; j++)
72  printf("f[%+2d] = %+5.3" __FES__ " %+5.3" __FES__ "*I\n",j,
73  creal(plan.f[j]), cimag(plan.f[j]));
74 
75  printf("\n");
76 
77  /* Fast approximate transformation, display result. */
78  nfsft_trafo(&plan);
79  printf("Vector f (NFSFT):\n");
80  for (j = 0; j < plan.M_total; j++)
81  printf("f[%+2d] = %+5.3" __FES__ " %+5.3" __FES__ "*I\n",j,
82  creal(plan.f[j]), cimag(plan.f[j]));
83 
84  printf("\n");
85 
86  /* Direct adjoint transformation, display result. */
87  nfsft_adjoint_direct(&plan);
88  printf("Vector f_hat (NDSFT):\n");
89  for (k = 0; k <= plan.N; k++)
90  for (n = -k; n <= k; n++)
91  fprintf(stdout,"f_hat[%+2d,%+2d] = %+5.3" __FES__ " %+5.3" __FES__ "*I\n",k,n,
92  creal(plan.f_hat[NFSFT_INDEX(k,n,&plan)]),
93  cimag(plan.f_hat[NFSFT_INDEX(k,n,&plan)]));
94 
95  printf("\n");
96 
97  /* Fast approximate adjoint transformation, display result. */
98  nfsft_adjoint(&plan);
99  printf("Vector f_hat (NFSFT):\n");
100  for (k = 0; k <= plan.N; k++)
101  {
102  for (n = -k; n <= k; n++)
103  {
104  fprintf(stdout,"f_hat[%+2d,%+2d] = %+5.3" __FES__ " %+5.3" __FES__ "*I\n",k,n,
105  creal(plan.f_hat[NFSFT_INDEX(k,n,&plan)]),
106  cimag(plan.f_hat[NFSFT_INDEX(k,n,&plan)]));
107  }
108  }
109 
110  /* Finalize the plan. */
111  nfsft_finalize(&plan);
112 
113  /* Destroy data precomputed for fast polynomial transform. */
114  nfsft_forget();
115 }

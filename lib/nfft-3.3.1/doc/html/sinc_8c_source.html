00021 R Y(sinc)(const R x)
00022 {
00023   /* Based on sinc function from Boost C++ library. */
00024   const R b =  EPSILON;
00025   const R bs = SQRT(b);
00026   const R bs2 = SQRT(bs);
00027 
00028   if (FABS(x) >= bs2)
00029     return SIN(x)/x;
00030   else
00031   {
00032     R r = K(1.0);
00033 
00034     if (FABS(x) >= b)
00035     {
00036       const R x2 = x * x;
00037       r -= x2 / K(6.0);
00038 
00039       if (FABS(x) >= bs)
00040         r += (x2 * x2) / K(120.0);
00041     }
00042 
00043     return r;
00044   }
00045 }
